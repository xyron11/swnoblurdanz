const express = require("express") const multer = require("multer") const axios = require("axios") const FormData = require("form-data") const fs = require("fs") const cors = require("cors") const app = express() app.use(cors()) app.use(express.json()) const upload = multer({ dest: "uploads/" }) let results = [] app.get("/", (req, res) => { res.send("API READY") }) app.get("/results", (req, res) => { res.json(results) results = [] }) app.post("/upload", upload.single("video"), async (req, res) => { try { const file = req.file const nomor = req.body.nomor if (!file) { return res.json({ status: false, error: "File tidak ditemukan" }) } if (!nomor) { fs.unlinkSync(file.path) return res.json({ status: false, error: "Nomor belum diisi" }) } const ext = file.originalname .split(".") .pop() .toLowerCase() const allowed = [ "mp4", "mov", "mkv", "webm", "avi", "m4v" ] if ( !allowed.includes(ext) ) { fs.unlinkSync(file.path) return res.json({ status: false, error: "Hanya file video" }) } const check = await axios.get( "https://raw.githubusercontent.com/USERNAME/REPO/main/verify.json" ) const members = check.data if ( !members.includes( nomor ) ) { fs.unlinkSync(file.path) return res.json({ status: false, error: "Nomor tidak ada di grup" }) } const job = await axios.post( "https://api.freeconvert.com/v1/process/jobs", { tasks: { import: { operation: "import/upload" }, convert: { operation: "convert", input: "import", input_format: ext, output_format: "mp4" }, "export-url": { operation: "export/url", input: "convert" } } } ) const data = job.data const uploadTask = data.tasks.find( v => v.name === "import" ) const uploadUrl = uploadTask.result.form.url const signature = uploadTask.result.form.parameters.signature const form = new FormData() form.append( "file", fs.createReadStream(file.path), file.originalname ) form.append( "signature", signature ) await axios.post( uploadUrl, form, { headers: { ...form.getHeaders() } } ) let resultUrl = null while (true) { const check = await axios.get( `https://api.freeconvert.com/v1/process/jobs/${data.id}` ) const exp = check.data.tasks.find( v => v.name === "export-url" ) if (exp.result?.url) { resultUrl = exp.result.url break } await new Promise(r => setTimeout(r, 3000) ) } fs.unlinkSync(file.path) results.push({ nomor, url: resultUrl }) res.json({ status: true, url: resultUrl }) } catch (e) { console.log(e) res.json({ status: false, error: e.message }) } }) app.listen( process.env.PORT || 3000 )
